dat <command> [--option <argument>]
  dat link <PATH>            creates a URL to share some data
  dat <URL> <PATH>           downloads and shares data from a dat link

options
  --quiet, -q         quiet output mode
  --debug             debug output mode
  --version, -v       get the installed dat version
  --port, -p          set a specific inbound tcp port
  --no-color          print output without colors
  --cwd               use a different current working directory
